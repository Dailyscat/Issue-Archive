# 변수 값 변경위한 매개변수 문법

```
${변수#패턴} // 변수에 설정된 문자열 앞에서부터 처음 찾은 패턴과 일치하는 패턴 앞의 모든 문자열 제거 
${변수##패턴} // 변수에 설정된 문자열 앞에서부터 마지막으로 찾은 패턴과 일치하는 패턴 앞의 모든 문자열 제거 
// 이때 패턴 앞에 모든 문자열을 의미하는 *가 필수적으로 들어가야한다.

FILE = "wow_containerwow.gd.gix"
echo ${FILE#*_} // containerwow~
```



```
${변수%패턴} // 변수에 설정된 문자열 뒤에서부터 처음 찾은 패턴과 일치하는 패턴 앞의 모든 문자열 제거 
${변수%%패턴} // 변수에 설정된 문자열 뒤에서부터 처음 찾은 패턴과 일치하는 패턴 앞의 모든 문자열 제거 
// 이때 패턴 뒤에 모든 문자열을 의미하는 *가 필수적으로 들어가야한다.

echo ${FILE%.*} // wow_containerwow.gd
echo ${FILE%%.*} // wow_containerwow
```

```
FILE_PATH="/etc/nova/nova.conf"
echo ${FILE_PATH##*/} // 마지막으로 찾은 /패턴 뒤의 문자열 nova.conf 출력
echo ${FILE_PATH%/*} // 뒤에서 처음으로 찾은 / 패턴 앞의 문자열 /etc/nova 출력

${#변수} // 변수 길이 리턴
echo ${#FILE_PATH}

```

```
${변수/찾을문자열/바꿀문자열} // 변수에서 처음으로 찾은 문자열만 바꿀문자열로 변경
${변수//찾을문자열/바꿀문자열} // 변수에서 처음부터 끝까지 찾은 모든 문자열을 바꿀 문자열로 변경

${변수/삭제할 문자열} // 변수에서 처음으로 찾은 문자열 삭제
${변수//삭제할 문자열} // 변수에서 처음부터 끝까지 찾은 모든 문자열을 바꿀 문자열로 변경

${변수/#찾을문자열/바꿀문자열} // 변수의 단어가 찾을문자열로 시작하면 해당 시작 부분만 바꿀 문자열로 변경
${변수/%찾을문자열/바꿀문자열} // 변수의 단어가 찾을문자열로 끝나면 해당 끝나는 부분만 바꿀 문자열로 변경

```
